"""

# ============================================================================
# ЧАСТЬ 1: БАЗОВЫЙ ПРИМЕР (ПРОИЗВОДСТВЕННАЯ ЗАДАЧА)

print("=" * 70)
print("ЧАСТЬ 1: РЕШЕНИЕ ПРОИЗВОДСТВЕННОЙ ЗАДАЧИ")
print("=" * 70)

# Прямая задача:
# F = 50x₁ + 40x₂ → max
# 3x₁ + 5x₂ ≤ 45 (ресурс 1)
# 4x₁ + 2x₂ ≤ 32 (ресурс 2)
# x₁, x₂ ≥ 0

# Коэффициенты целевой функции (для linprog нужна минимизация, поэтому берем с минусом)
c = [-50, -40]

# Матрица ограничений и правые части (все неравенства типа ≤)
A_ub = [[3, 5], [4, 2]]
b_ub = [45, 32]

# Границы переменных (неотрицательные)
bounds = [(0, None), (0, None)]

# Решение прямой задачи
res_primal = linprog(c, A_ub=A_ub, b_ub=b_ub, bounds=bounds, method='highs')

print("\n📊 РЕЗУЛЬТАТЫ ПРЯМОЙ ЗАДАЧИ:")
print(f"  Статус: {res_primal.message}")
print(f"  Оптимальный план: x₁ = {res_primal.x[0]:.4f}, x₂ = {res_primal.x[1]:.4f}")
print(f"  Максимальная прибыль: F_max = {-res_primal.fun:.4f}")

# Использование ресурсов
print("\n  Использование ресурсов:")
for i in range(len(b_ub)):
    used = A_ub[i][0] * res_primal.x[0] + A_ub[i][1] * res_primal.x[1]
    slack = res_primal.slack[i]
    print(f"    Ресурс {i+1}: {used:.4f} / {b_ub[i]} (остаток: {slack:.4f})")

# y₁, y₂ ≥ 0

# 3y₁ + 4y₂ ≥ 50  →  -3y₁ - 4y₂ ≤ -50
# 5y₁ + 2y₂ ≥ 40  →  -5y₁ - 2y₂ ≤ -40

    product = res_dual.x[i] * slack_i
    print(f"    y_{i+1} * (b_{i+1} - A_{i+1}·x) = {res_dual.x[i]:.4f} * {slack_i:.4f} = {product:.6f}")

    if abs(product) < 1e-6:
        print(f"      ✅ Условие выполняется (x_{j+1} = 0 или ограничение активно)")

def build_dual_symmetric(c, A, b):
    """
    Построение двойственной задачи для симметричного случая
    
    Прямая задача (max):
        max c^T x
        A x ≤ b
        x ≥ 0
    
    Двойственная задача (min):
        min b^T y
        A^T y ≥ c
        y ≥ 0
    
    Параметры:
    ----------
    c : list или array
        Коэффициенты целевой функции прямой задачи (для max)
    A : list или array
        Матрица ограничений
    b : list или array
        Правые части ограничений
    
    Возвращает:
    -----------
    c_dual : array
        Коэффициенты целевой функции двойственной задачи
    A_ub_dual : array
        Матрица ограничений для linprog (приведена к виду ≤)
    b_ub_dual : array
        Правые части для linprog
    bounds_dual : list
        Границы переменных двойственной задачи
    """
    c_dual = np.array(b, dtype=float)
    
    # Исходно: A^T y ≥ c
    # Приводим к виду ≤: -A^T y ≤ -c
    A_ub_dual = -np.array(A, dtype=float).T
    b_ub_dual = -np.array(c, dtype=float)
    
    bounds_dual = [(0, None)] * len(b)
    
    return c_dual, A_ub_dual, b_ub_dual, bounds_dual

def solve_and_analyze(c, A, b, problem_type='max'):
    """
    Полное решение и анализ задачи ЛП с проверкой двойственности
    
    Параметры:
    ----------
    c : list или array
        Коэффициенты целевой функции
    A : list или array
        Матрица ограничений (неравенства ≤)
    b : list или array
        Правые части ограничений
    problem_type : str
        'max' или 'min'
    
    Возвращает:
    -----------
    results : dict
        Словарь с результатами анализа
    """
    print(f"\n{'='*60}")
    print(f"АНАЛИЗ ЗАДАЧИ ЛП ({problem_type.upper()})")
    print(f"{'='*60}")
    
    print("\nИсходные данные:")
    print(f"  Целевая функция: {'max' if problem_type == 'max' else 'min'} Z = ", end="")
    for i, ci in enumerate(c):
        print(f"{ci:+}x_{i+1} ", end="")
    print()
    print(f"  Ограничения:")
    for i, (row, bi) in enumerate(zip(A, b)):
        print(f"    ", end="")
        for j, aij in enumerate(row):
            print(f"{aij:+}x_{j+1} ", end="")
        print(f"≤ {bi}")
    
    # Решение прямой задачи
    if problem_type == 'max':
        c_linprog = [-ci for ci in c]
    else:
        c_linprog = c
    
    res_primal = linprog(c_linprog, A_ub=A, b_ub=b, bounds=[(0, None)]*len(c), method='highs')
    
    if not res_primal.success:
        print(f"\n❌ Ошибка: {res_primal.message}")
        return None
    
    if problem_type == 'max':
        primal_value = -res_primal.fun
    else:
        primal_value = res_primal.fun
    
    print(f"\n📊 Решение прямой задачи:")
    for i, xi in enumerate(res_primal.x):
        print(f"    x_{i+1}* = {xi:.6f}")
    print(f"    Z* = {primal_value:.6f}")
    
    # Построение и решение двойственной задачи
    c_dual, A_dual, b_dual, bounds_dual = build_dual_symmetric(c, A, b)
    res_dual = linprog(c_dual, A_ub=A_dual, b_ub=b_dual, bounds=bounds_dual, method='highs')
    
    print(f"\n📊 Решение двойственной задачи:")
    for i, yi in enumerate(res_dual.x):
        print(f"    y_{i+1}* = {yi:.6f}")
    print(f"    W* = {res_dual.fun:.6f}")
    
    # Проверка теорем
    print(f"\n📐 Проверка теорем двойственности:")
    print(f"    F_max = {primal_value:.6f}")
    print(f"    Z_min = {res_dual.fun:.6f}")
    
    if abs(primal_value - res_dual.fun) < 1e-6:
        print("    ✅ Теорема 1 выполняется: значения совпадают")
    else:
        print(f"    ❌ Теорема 1 НЕ выполняется: разница = {abs(primal_value - res_dual.fun):.10f}")
    
    return {
        'primal_solution': res_primal.x,
        'primal_value': primal_value,
        'dual_solution': res_dual.x,
        'dual_value': res_dual.fun,
        'status': 'success'
    }
